{

std::iota(std::begin(v), std::end(v), mt);
std::random_device seed;
std::mt19937 mt(seed());
std::shuffle(std::begin(v), std::end(v), mt);
TPLCXX17::chap16_7_1::v1::insertion_sort(std::begin(v), std::end(v), std::less<>(), v2::binary_search_insert()); // less than

}
