592 ForwardIterator lower_bound(ForwardIterator first, ForwardIterator last, const T& val, Compare comp)
593 {
594  typedef typename std::iterator_traits<ForwardIterator>::difference_type diff_type;
595 
596  for (diff_type len = std::distance(first, last); len > 0;) {
597  diff_type half = len >> 1; // == len / 2
598  ForwardIterator mid = first;
599  std::advance(mid, half);
600 
601  if (comp(*mid, val)) {
602  first = mid;
603  ++first;
604  len -= half - 1;
605  } else {
606  len = half;
607  }
608  }
609  return first;
610 }
